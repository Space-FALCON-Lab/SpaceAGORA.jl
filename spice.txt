 $INPUT
  SpicePath = 'GRAM_Data/SPICE'
  SpiceLsk = '/lsk/naif0012.tls'
 $END                

 Explanation of variables:
  SpicePath    = Path to NAIF Spice data.
  SpiceLsk     = Optional override of the relative (to SpicePath) path to the Spice leapseconds LSK file.
  SpicePck     = Optional override of the relative (to SpicePath) path to the Spice planetary constants PCK file.
  SpiceVenus   = Optional override of the relative (to SpicePath) path to the Spice Venus kernel.
  SpiceEarth   = Optional override of the relative (to SpicePath) path to the Spice Earth kernel.
  SpiceMars    = Optional override of the relative (to SpicePath) path to the Spice Mars kernel.
  SpiceJupiter = Optional override of the relative (to SpicePath) path to the Spice Jupiter kernel.
  SpiceSaturn  = Optional override of the relative (to SpicePath) path to the Spice Saturn kernel.
  SpiceUranus  = Optional override of the relative (to SpicePath) path to the Spice Uranus kernel.
  SpiceNeptune = Optional override of the relative (to SpicePath) path to the Spice Neptune kernel.
  SpiceTitan   = Optional override of the relative (to SpicePath) path to the Spice Titan kernel.
